// stores/library.ts
import { ref, watch } from "vue"

const queryInput = ref("")         // 用户实时输入
const queryDebounced = ref("")     // 300ms 后才生效

let timer: number | undefined
watch(queryInput, (v) => {
  if (timer) clearTimeout(timer)
  timer = window.setTimeout(() => {
    queryDebounced.value = v
  }, 300)
})

// LibraryView 仍然 bind v-model 到 queryInput
// store.load 读 queryDebounced

## Task 2: circle tag chip 栏

**Files:**
- Modify: `src/views/LibraryView.vue`

- [ ] **Step 1: store 加 `topCircles` getter**

```typescript
// stores/library.ts
const topCircles = computed(() => {
  const counts = new Map<string, number>()
  for (const f of items) {
    if (f.circle) counts.set(f.circle, (counts.get(f.circle) ?? 0) + 1)
  }
  return Array.from(counts.entries())
    .sort((a, b) => b[1] - a[1])
    .slice(0, 10)
    .map(([name, count]) => ({ name, count }))
})
```

- [ ] **Step 2: LibraryView 加 chip 栏**

```vue
<template>
  <div>
    <div class="page-header">...</div>
    <n-space style="margin-bottom: 16px">
      <n-input ... />
      <n-select ... />
    </n-space>
    <n-space v-if="store.topCircles.length > 0" style="margin-bottom: 16px">
      <n-tag
        v-for="c in store.topCircles"
        :key="c.name"
        :type="store.getQuery() === c.name ? 'primary' : 'default'"
        checkable
        @click="store.setQuery(c.name)"
      >
        {{ c.name }} ({{ c.count }})
      </n-tag>
      <n-tag v-if="store.getQuery()" type="warning" @click="store.setQuery('')">
        清除
      </n-tag>
    </n-space>
    <n-spin :show="store.loading">...</n-spin>
  </div>
</template>
```
